# Raspberry Pi 5 bare-metal boot configuration.
arm_64bit=1
kernel=kernel_2712.img
kernel_address=0x80000
disable_commandline_tags=1

# Enable the RP1 PL011 on GPIO14/GPIO15 and print firmware UART logs too.
enable_rp1_uart=1
enable_uart=1
uart_2ndstage=1

# Keep the RP1 PCIe link alive for the kernel. Without this, RP1 MMIO
# at the firmware-reported 0x1c00000000 handoff window is inaccessible.
pciex4_reset=0

# Pi 5 active cooler: request full fan speed from the first thermal level.
dtparam=cooling_fan=on
dtparam=fan_temp0=0
dtparam=fan_temp0_hyst=0
dtparam=fan_temp0_speed=255
