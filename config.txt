# config.txt  —  Raspberry Pi 5 bare-metal boot configuration

# Load our 64-bit bare-metal kernel
arm_64bit=1
kernel=kernel8.img

# kernel8.img is loaded to 0x80000 by default in 64-bit mode.
# Uncomment the line below only if you rename the image.
# kernel_address=0x80000

# Prevent the firmware from appending a Linux boot command line
# or device-tree tags that would clobber memory near 0x80000.
disable_commandline_tags=2

# Keep all 4 cores at their reset state; core 0 is released to
# our kernel automatically.  Cores 1-3 sit in the firmware
# spin-table until explicitly woken (we park them in boot.S).
# kernel_old=0  (default — do not set kernel_old=1 for bare metal)
